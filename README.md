## Install

```bash
npm install --save react-mui-notify
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'mui-notify'
import 'mui-notify/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
